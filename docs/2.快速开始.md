# 快速开始

## 安装

```bash
pip install autestoy
```

或使用 uv：

```bash

```

## 基本用法

### 1. 初始化

```python
import autestoy as att

# v0.2.0 起需要显式初始化
att.init()
# 终端输出: [INFO] Script start at [2026-01-01 12:00:00.000]
```

### 2. SSH 远程执行

```python
import autestoy as att

att.init()

# 配置远程主机
conf = att.RemoteConfig(
    user="root",
    ip="192.168.1.100",
    password="password",
    port=22,
).set_name("DUT")  # 设置别名，影响终端显示

# 连接并执行命令
dut = att.SSH(conf)

# 单条命令
record = dut.exec_run("uname -a")
# 终端输出:
# [   0.532] [1][DUT] $ uname -a
# [   0.589] Linux dut 6.1.0 armv7l GNU/Linux

# 检查输出
if "Linux" in record:
    print("确认是 Linux 系统")

# 提取字段（类似 cut 命令）
record = dut.exec_run("ps")
pid_list = record.cut_fields(1, re_delimiter=r"\s+")

# 正则搜索
record = dut.exec_run("ls -l")
if match := record.search(r"log_(\d+)\.txt"):
    print(f"找到日志文件: {match.group(0)}")
```

### 3. 路径管理

SSH 提供三种路径管理方式：

```python
dut = att.SSH(conf)

# (a) 临时路径 — 仅对下一条命令生效
dut.with_path("/home/project").exec_run("ls")

# (b) 全局路径 — 所有后续命令在此路径下执行
dut.set_global_path("./build")
dut.exec_run("make")       # 等同于 cd ./build && make
dut.exec_run("make test")  # 等同于 cd ./build && make test

# (c) cd 命令 — 改变全局路径
dut.cd("/home/project/release")
dut.exec_run("pwd")
# /home/project/release
```

### 4. 顺序执行多条命令

```python
dut = att.SSH(conf)

# 多行字符串（自动跳过空行和 # 注释行）
dut.exec_run_lines("""
cd /home/project
git pull
make clean
make
""")

# 列表
dut.exec_run_lines(["./configure", "make", "make install"])

# 混用
dut.exec_run_lines("./step1.sh", "./step2.sh", "./step3.sh")
```

### 5. 后台长时间运行 + 实时处理输出

```python
dut = att.SSH(conf)

# 在后台运行一个持续输出的命令
tail_task = dut.long_running("tail -f /var/log/syslog")

# 触发一些操作
dut.exec_run("systemctl restart some-service")

# 实时检查输出
timeout = att.TrySeconds(10)
while timeout:
    result = tail_task.search_next_line(r"error|fail")
    if result:
        line, match = result
        att.ulog(f"检测到异常: {line}")
        break

# 结束后台任务
tail_task.task_kill()
```

### 6. 交互式 Channel（保持上下文）

```python
dut = att.SSH(conf)

# 创建交互式通道
ch = dut.create_channel(name="shell")

# 通道内保持目录和状态
ch.run("cd /home/project")
ch.run("pwd")
# /home/project

ch.run("ls -la")
ch.run("./my_script.sh arg1 arg2")

if "FAIL" in ch.last_record:
    att.ulog("测试失败")
```

### 7. SFTP 文件传输

```python
dut = att.SSH(conf)
sftp = dut.create_ftp()

sftp.put("local_file.bin", "/remote/path/file.bin")
sftp.get("/remote/path/log.txt", "local_log.txt")

# 自动创建远程目录
sftp.put("data.bin", "/remote/new_dir/data.bin", auto_mkdir=True)
```

### 8. 本机命令执行

```python
local = att.Local()
record = local.run("df -h")
record = local.run("uname -a")
```

### 9. 数据类型：Bits

`Bits` 是类 Verilog 的位宽数据类型，适合嵌入式/硬件测试：

```python
# 从整数创建
reg_val = att.Bits(0xA5, 8)
print(reg_val)        # 0xa5
print(f"{reg_val:b}") # 10100101

# 从字符串创建（支持 Verilog 风格）
val = att.Bits("8'd255")   # 8 位宽，值 255
val = att.Bits("16'hFF00") # 16 位宽，值 0xFF00
val = att.Bits("32'o377")  # 32 位宽，八进制

# 切片访问（类似 Verilog）
val = att.Bits(0x12345678, 32)
low_byte = val[7:0]         # 0x78
high_word = val[31:16]      # 0x1234

# 切片赋值
reg = att.Bits(0, 32)
reg[15:0] = 0xA5A5
reg[31:16] = 0x5A5A

# 位拼接
a = att.Bits(0xFF, 8)
b = att.Bits(0x00, 8)
c = a @ b      # 0xFF00, 16 位

# 字段借用（与原始 Bits 同步修改）
view = att.BitView(reg, 15, 0)
view.value = 0x1234
print(reg)  # 低 16 位已同步修改
```

### 10. 用户日志

```python
att.ulog("这是一个普通日志")

att.ulog("关键步骤完成", override_font_color=att.AnsiColor.green)

att.ulog("警告：磁盘空间不足",
         override_font_color=att.AnsiColor.yellow)
```

### 11. 超时控制

```python
# 尝试 5 秒内等待某个条件
timer = att.TrySeconds(5)
while timer:
    if condition_met():
        att.ulog("条件满足")
        break
else:
    att.ulog("超时")
```

## 完整示例：自动化测试脚本

```python
import autestoy as att

att.init()

# ===== 连接设备 =====
dut = att.RemoteConfig(
    user="root", ip="192.168.1.100",
    password="password", port=22,
).set_name("DUT")
ssh = att.SSH(dut)

try:
    # ===== 检查环境 =====
    uname = ssh.exec_run("uname -m")
    if "arm" not in uname:
        att.ulog("不是 ARM 平台，跳过", att.AnsiColor.yellow)
        exit(0)

    # ===== 编译 =====
    ssh.exec_run_lines("""
    cd /home/project
    git pull
    make clean
    make -j4
    """)

    # ===== 运行测试 =====
    test_task = ssh.long_running("./run_tests.sh --verbose")

    # 实时监控输出
    timeout = att.TrySeconds(120)
    passed = failed = 0
    while timeout:
        result = test_task.search_next_line(r"(PASS|FAIL)")
        if result:
            line, match = result
            if match.group(1) == "PASS":
                passed += 1
            else:
                failed += 1

    test_task.task_kill()
    att.ulog(f"测试结果: {passed} 通过, {failed} 失败")

finally:
    ssh.close_all()
```
